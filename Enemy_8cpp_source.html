181
182 if (duration_cast<milliseconds>(timer - m_moveTimer).count() >

198 }

221// Bresenham's line algorithm
231std::vector<std::pair<int, int>> get_tiles_on_line(std::pair<int, int> tile1,
232 std::pair<int, int> tile2) {
233 std::vector<std::pair<int, int>> tiles;
234 int x1 = tile1.first, y1 = tile1.second, x2 = tile2.first,
235 y2 = tile2.second;
236 int dx = abs(x2 - x1), dy = abs(y2 - y1);
237 int sx = x1 < x2 ? 1 : -1, sy = y1 < y2 ? 1 : -1;
238 int err = dx - dy;
239
240 while (true) {
241 tiles.push_back(std::make_pair(x1, y1));
242 if (x1 == x2 && y1 == y2) break;
243 int e2 = 2 * err;
244 if (e2 > -dy) {
245 err -= dy;
246 x1 += sx;
247 }
248 if (e2 < dx) {
249 err += dx;
250 y1 += sy;
251 }
252 }
253 return tiles;
254}
