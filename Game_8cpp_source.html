108void Game::LoadLevel(UINT level) {

112 }
113 // clear enemies

115 m_map.removeCreature(m_enemies[i]->getId());
116 delete m_enemies[i];
117 }
118 m_enemies.clear();
119
120 // load new map
121 m_map.loadMap(allLevels[level].getMap());
122 m_player.setX(allLevels[level].getStartingPosition().first);
123 m_player.setY(allLevels[level].getStartingPosition().second);
124 m_map.addCreature(0, &m_player);
125
126 // clear creatures id counter
127 m_lastId = 0;
128
129 for (size_t i = 0; i < allLevels[level].getEnemies().size(); i++) {
130 Enemy* newEnemy = allLevels[level].getEnemies()[i];
131
132 int newEnemyId = AddEnemy(newEnemy->getX(), newEnemy->getY());

140 }

452void Game::Update() {
453 char32_t forbiddenTile = U'█';
454 int prevX = m_player.getX();
455 int prevY = m_player.getY();
456
457 // allow player to move only m_moveTimer ticks speed times
458
459 // get input
460 if (GetAsyncKeyState(VK_UP) || GetAsyncKeyState(0x57)) {
461 // check if player can move up
462 if (m_player.getY() > 0) {
463 m_player.moveUp();
464 }
465 } else if (GetAsyncKeyState(VK_DOWN) || GetAsyncKeyState(0x53)) {
466 if (m_player.getY() < m_map.getHeight() - 1) {
467 m_player.moveDown();
468 }
469 } else if (GetAsyncKeyState(VK_LEFT) || GetAsyncKeyState(0x41)) {
470 if (m_player.getX() > 1) {
471 m_player.moveLeft();
472 }
473 } else if (GetAsyncKeyState(VK_RIGHT) || GetAsyncKeyState(0x44)) {
474 if (m_player.getX() < m_map.getWidth() - 3) {
475 m_player.moveRight();
476 }
477 }
478 if (GetAsyncKeyState(VK_SPACE)) {
479 bool wasAttacked = false;
480 // find m_enemies in m_player.getAttackRange
481 for (int i = -m_player.getAttackRange() * 2;
482 i <= m_player.getAttackRange() * 2; i++) {
483 int foundEnemy = FindEnemy(m_player.getX() + i, m_player.getY());
484 if (foundEnemy != -1) {
485 // attack enemy
486 wasAttacked = true;
487 m_player.attack(*m_enemies[foundEnemy]);
488 bool isDead = m_enemies[foundEnemy]->getHealth() <= 0;
489
490 if (isDead) {
491 // add exp
492 m_player.setExperience(
493 m_player.getExperience() +
494 m_enemies[foundEnemy]->getExperience());
495 // enemy is dead
496 DropRandomItem(m_enemies[foundEnemy]->getX(),
497 m_enemies[foundEnemy]->getY());
498 RemoveEnemy(m_enemies[foundEnemy]->getId());
499 }
500 }
501 }
502 for (int i = -m_player.getAttackRange(); i <= m_player.getAttackRange();
503 i++) {
504 int foundEnemy = FindEnemy(m_player.getX(), m_player.getY() + i);
505 if (foundEnemy != -1) {
506 // attack enemy
507 wasAttacked = true;
508 m_player.attack(*m_enemies[foundEnemy]);
509 bool isDead = m_enemies[foundEnemy]->getHealth() <= 0;
510
511 if (isDead) {
512 // add exp
513 m_player.setExperience(
514 m_player.getExperience() +
515 m_enemies[foundEnemy]->getExperience());
516 // enemy is dead
517 DropRandomItem(m_enemies[foundEnemy]->getX(),
518 m_enemies[foundEnemy]->getY());
519 RemoveEnemy(m_enemies[foundEnemy]->getId());
520 }
521 }
522 }
523 if (!wasAttacked) {
524 m_player.attack();
525 }
526 }
527
528 // if (GetAsyncKeyState((int)'E')) {
529 // // increase experience
530 // m_player.setExperience(m_player.getExperience() + 1);
531 // }
532
533 if (isOnTile(m_player, forbiddenTile) || isOnEnemy(m_player)) {
534 // if so, move player back to previous position
535 m_player.setX(prevX);
536 m_player.setY(prevY);
537 }
538
539 // Pick up item
540 if (m_map.getItem(m_player.getX(), m_player.getY()) != nullptr) {
541 Item* pickedItem = m_map.getItem(m_player.getX(), m_player.getY());
542 if (pickedItem->getType() == ItemType::Food) {
543 m_player.setHealth(m_player.getHealth() + pickedItem->getValue());
544 }
545 if (pickedItem->getType() == ItemType::Weapon) {
546 m_player.setAttackPower(m_player.getAttackPower() +
547 pickedItem->getValue());
548 }
549 if (pickedItem->getType() == ItemType::RangedWeapon) {
550 m_player.setAttackRange(m_player.getAttackRange() +
551 pickedItem->getValue());
552 }
553 if (pickedItem->getType() == ItemType::Experience) {
554 m_player.setExperience(m_player.getExperience() +
555 pickedItem->getValue());
556 }
557 m_player.addToInventory(pickedItem);
558
559 m_map.removeItem(pickedItem);
560 }
561
562 // update position of visible melee weapons
563 // if timer for attack is below attack speed /2
564
565 if (duration_cast<milliseconds>(steady_clock::now() -
566 m_player.getAttackTimer())
567 .count() >= m_player.getAttackSpeed() / 4) {
568 m_player.setIsMeleeAttack(false);
569 }
570 // add them to the map
571
572 // exit game on escape
573 if (GetAsyncKeyState(VK_ESCAPE)) {
574 m_isRunning = false;
575 }
576
577 // if all enemies are dead, show exit
578 if (m_enemies.size() == 0) {
579 // add exit to map
580 m_map.setTileAppearance(
581 allLevels[m_currentLevel].getExitPosition().first,
582 allLevels[m_currentLevel].getExitPosition().second, U'🚪');
583 // if player on exit load next level
584 if ((m_player.getX() ==
585 allLevels[m_currentLevel].getExitPosition().first ||
586 m_player.getX() + 1 ==
587 allLevels[m_currentLevel].getExitPosition().first) &&
588 m_player.getY() ==
589 allLevels[m_currentLevel].getExitPosition().second) {
590 // load next level
591 m_currentLevel++;
592 LoadLevel(m_currentLevel);
593 }
594 }
595
596 // update position of enemies
597 for (size_t i = 0; i < m_enemies.size(); i++) {
598 if (m_enemies[i]->checkLineOfSight(m_player, m_map)) {
599 m_enemies[i]->moveTowardsPlayer(m_player);
600 }
601 }
602
603 // check if player is dead
604 if (m_player.getHealth() <= 0) {
605 m_isRunning = false;
606 }
607}

612void Game::PreRender() {
613 // clear hidden buffer
614 for (int i = 0; i < SCREEN_WIDTH * SCREEN_HEIGHT; i++) {
615 hiddenBuffer[i].Char.UnicodeChar = ' ';
616 hiddenBuffer[i].Attributes = FOREGROUND_GREEN | FOREGROUND_BLUE;
617 }
618 // get stats, map, inventory
619 std::wstring stats = showStats();
620 std::u32string map = m_map.getMapString();
621 std::string inventory = showInventory();
622
623 // add stats to hidden buffer
624 int x = 0;
625 int y = 0;
626 for (UINT i = 0; i < stats.length(); i++) {
627 if (stats[i] == '\t') {
628 for (UINT j = 0; j < 4; j++) {
629 hiddenBuffer[x + SCREEN_WIDTH * y].Char.UnicodeChar = ' ';
630 hiddenBuffer[x + SCREEN_WIDTH * y].Attributes =
631 FG_COLORS::FG_GREEN;
632 x += 1;
633 }
634 } else if (stats[i] == '\n') {
635 x = 0;
636 y += 1;
637 } else {
638 hiddenBuffer[x + SCREEN_WIDTH * y].Char.UnicodeChar = stats[i];
639 hiddenBuffer[x + SCREEN_WIDTH * y].Attributes = FG_COLORS::FG_GREEN;
640 x += 1;
641 }
642 }
643
644 x = 0;
645 y = 2;
646 // add map to hidden buffer
647 for (UINT i = 0; i < map.length(); i++) {
648 if (map[i] == '\n') {
649 x = 0;
650 y += 1;
651 } else {
652 // if map[i] bigger than wchar size, split into two wchars
653 bool isSurrogatePair;
654 std::wstring wstr = char32to16(map[i], &isSurrogatePair);
655
656 if (isSurrogatePair) {
657 i += 1;
658 }
659
660 // add bytes to hidden buffer
661 for (UINT j = 0; j < wstr.length(); j++) {
662 hiddenBuffer[x + SCREEN_WIDTH * y].Char.UnicodeChar = wstr[j];
663 hiddenBuffer[x + SCREEN_WIDTH * y].Attributes =
664 FG_COLORS::FG_WHITE;
665 x += 1;
666 }
667 }
668 }
669 // draw swords if player is attacking
670 if (m_player.getIsMeleeAttack()) {
671 std::vector<Creature*> attacks = m_player.getMeleeWeaponList();
672 for (UINT i = 0; i < attacks.size(); i++) {
673 bool isSurrogatePair;
674 // check if
675 std::wstring wstr =
676 char32to16(attacks[i]->getAppearance(), &isSurrogatePair);

678 for (UINT j = 0; j < wstr.length(); j++) {
679 UINT coord = attacks[i]->getX() + j + 3 +
680 SCREEN_WIDTH * (attacks[i]->getY() + 4);

682 hiddenBuffer[coord].Attributes = FG_COLORS::FG_WHITE;
683 }
684 }
685 }
686
687 // inventory no use
688 // return;
689
690 x = 0;
691 y = m_map.getHeight() + 6;
692 // add inventory to hidden buffer
693 for (UINT i = 0; i < inventory.length(); i++) {
694 if (inventory[i] == '\t') {
695 for (int j = 0; j < 4; j++) {
696 hiddenBuffer[x + SCREEN_WIDTH * y].Char.UnicodeChar = ' ';
697 hiddenBuffer[x + SCREEN_WIDTH * y].Attributes =

706 hiddenBuffer[x + SCREEN_WIDTH * y].Attributes = FG_COLORS::FG_BROWN;
